#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;
/*

* 문제 A : 최댓값 최솟값 동시에 찾기
* 배열에서 비교하지 않은 두개의 요소를 비교한 후 각각의 max, min들과 비교
* 그 후 최대 최소를 정함
* 
*/
void FindMaxMin(std::vector<int>& vec) {
    int max = 0, min = 0;
    int idx = 0;

    if (vec.size() % 2 == 1) {
        max = min = vec[0];
        idx = 1;
    }
    else {
        if (vec[0] < vec[1]) {
            max = vec[1];
            min = vec[0];
        }
        else {
            max = vec[0];
            min = vec[1];
        }
        idx = 2;
    }

    while (idx <= vec.size() - 1) {
        if (vec[idx] < vec[idx + 1]) {  //두개의 비교하지 않은 요소들을 비교해서 각각의 winner, loser 정하기 
            if (max < vec[idx + 1]) max = vec[idx + 1];
            if (min > vec[idx]) min = vec[idx];
        }
        else {
            if (max < vec[idx]) max = vec[idx];
            if (min > vec[idx + 1]) min = vec[idx + 1];
        }

        idx += 2;   //다음 두개의 비교하지 않은 요소들을 비교
    }

    std::cout << max << " " << min << "\n";
}

void TestCase() {
    int n = 0;
    std::cin >> n;

    std::vector<int> vec(n);

    for (int i = 0; i < n; i++) {
        std::cin >> vec[i];
    }

    FindMaxMin(vec);
}

int main(void) {
    ios::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);

    int t{ 0 };
    std::cin >> t;
    for (int i = 0; i < t; i++)
        TestCase();

    return 0;
}
